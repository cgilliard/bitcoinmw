Expand description

Build the specified crate::ErrorKind and convert it into an crate::Error. The desired crate::ErrorKind is specified using the crate::ErrKind name enum.

Example:

 use bmw_err::{Error, ErrorKind, ErrKind, err};

 fn main() -> Result<(), Error> {
     show_err_kind(false)?;
     Ok(())
 }

 fn show_err_kind(do_error: bool) -> Result<(), Error> {
     let e = err!(ErrKind::Configuration, "invalid parameter name");

     if do_error {
         return Err(e);
     }

     Ok(())
 }
